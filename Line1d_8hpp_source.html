354 inline void System::computeForcePotential()
355 {
356  for (size_t p = 0; p < m_N; ++p) {
357  double x = m_x(p);
358  m_y[p].set_x(x);
359  m_f_potential(p) = m_mu * (0.5 * (m_y[p].yright() + m_y[p].yleft()) - x);
360  }
361 }
362 
363 inline void System::computeForceNeighbours()
364 {
365  for (size_t p = 1; p < m_N - 1; ++p) {
366  m_f_neighbours(p) = m_k_neighbours * (m_x(p - 1) - 2 * m_x(p) + m_x(p + 1));
367  }
368  m_f_neighbours.front() = m_k_neighbours * (m_x.back() - 2 * m_x.front() + m_x(1));
369  m_f_neighbours.back() = m_k_neighbours * (m_x(m_N - 2) - 2 * m_x.back() + m_x.front());
370 }

387 inline void System::timeStep()
388 {
389  m_t += m_dt;
390  xt::noalias(m_v_n) = m_v;
391  xt::noalias(m_a_n) = m_a;
392 
393  // positions
394  xt::noalias(m_x) = m_x + m_dt * m_v + 0.5 * std::pow(m_dt, 2.0) * m_a;
395  this->computeForcePotential();
396  this->computeForceNeighbours();
397  this->computeForceFrame();
398 
399  // velocities
400  xt::noalias(m_v) = m_v_n + m_dt * m_a_n;
401  this->computeForceDamping();
402 
403  // accelerations
404  this->computeForce();
405  xt::noalias(m_a) = m_f / m_m;
406 
407  // velocities
408  xt::noalias(m_v) = m_v_n + 0.5 * m_dt * (m_a_n + m_a);
409  this->computeForceDamping();
410 
411  // accelerations
412  this->computeForce();
413  xt::noalias(m_a) = m_f / m_m;
414 
415  // velocities
416  xt::noalias(m_v) = m_v_n + 0.5 * m_dt * (m_a_n + m_a);
417  this->computeForceDamping();
418 
419  // accelerations
420  this->computeForce();
421  xt::noalias(m_a) = m_f / m_m;
422 
423  if (xt::any(xt::isnan(m_x))) {
424  throw std::runtime_error("NaN entries found");
425  }
426 }

453 inline size_t System::timeStepsUntilEvent(double tol, size_t niter_tol, size_t max_iter)
454 {
455  GooseFEM::Iterate::StopList stop(niter_tol);
456 
457  auto i_n = this->i();
458 
459  for (size_t iiter = 1; iiter < max_iter; ++iiter) {
460 
461  this->timeStep();
462 
463  for (size_t p = 0; p < m_N; ++p) {
464  if (m_y[p].i() != i_n(p)) {
465  return iiter;
466  }
467  }
468 
469  if (stop.stop_simple(this->residual(), tol)) {
470  this->quench();
471  return 0;
472  }
473  }
474 
475  throw std::runtime_error("No convergence found");
476 }

495 inline size_t System::minimise_timeactivity(double tol, size_t niter_tol, size_t max_iter)
496 {
497  GooseFEM::Iterate::StopList stop(niter_tol);
498 
499  auto i_n = this->i();
500  long s = 0;
501  long s_n = 0;
502  size_t first_iter = 0;
503  size_t last_iter = 0;
504  bool init = true;
505 
506  for (size_t iiter = 0; iiter < max_iter; ++iiter) {
507 
508  this->timeStep();
509 
510  s = 0;
511 
512  for (size_t p = 0; p < m_N; ++p) {
513  s += std::abs(m_y[p].i() - i_n(p));
514  }
515 
516  if (s != s_n) {
517  if (init) {
518  init = false;
519  first_iter = iiter;
520  }
521  last_iter = iiter;
522  }
523 
524  s_n = s;
525 
526  if (stop.stop(this->residual(), tol)) {
527  this->quench();
528  return last_iter - first_iter;
529  }
530  }
531 
532  throw std::runtime_error("No convergence found");
533 }

569 inline void System::advanceElastic(double dx, bool dx_of_frame)
570 {
571  double dx_particles;
572  double dx_frame;
573 
574  if (dx_of_frame) {
575  dx_frame = dx;
576  dx_particles = dx * m_k_frame / (m_k_frame + m_mu);
577  }
578  else {
579  dx_particles = dx;
580  dx_frame = dx * (m_k_frame + m_mu) / m_k_frame;
581  }
582 
583  #ifdef FRICTIONQPOTSPRINGBLOCK_ENABLE_DEBUG
584  if (dx_particles > 0.0) {
585  FRICTIONQPOTSPRINGBLOCK_DEBUG(dx_particles < xt::amin(this->yieldDistanceRight())());
586  }
587  else {
588  FRICTIONQPOTSPRINGBLOCK_DEBUG(std::abs(dx_particles) < xt::amin(this->yieldDistanceLeft())());
589  }
590  #endif
591 
592  m_x += dx_particles;
593  m_x_frame += dx_frame;
594  this->computeForcePotential();
595  this->computeForceNeighbours();
596  this->computeForceFrame();
597  this->computeForce();
598 }
